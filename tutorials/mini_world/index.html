    2. The Code Structure

    Below is the annotated version of the experimentâ€™s logic. You can paste this into a Jupyter notebook or run directly.

    from dsg_jit.world.scene_graph import SceneGraphWorld
from dsg_jit.world.visualization import plot_factor_graph_2d
import jax.numpy as jnp

# Create a new world
sg = SceneGraphWorld()

# Add a single room at the origin
roomA = sg.add_room3d(jnp.array([0.0, 0.0, 0.0]))

# Add two places inside the room
placeA = sg.add_place3d(roomA, jnp.array([1.0, 0.0, 0.0]))
placeB = sg.add_place3d(roomA, jnp.array([-1.0, 0.0, 0.0]))

# Add objects inside each place
objA1 = sg.add_object3d(placeA, jnp.array([1.2, 0.1, 0.0]))
objA2 = sg.add_object3d(placeA, jnp.array([0.8, -0.2, 0.0]))
objB1 = sg.add_object3d(placeB, jnp.array([-1.2, 0.2, 0.0]))

# Access the underlying WorldModel (factor graph + residual architecture)
wm = sg.wm

# Visualize the factor graph structure
plot_factor_graph_2d(wm.fg, title="Mini-World Scene Graph")
